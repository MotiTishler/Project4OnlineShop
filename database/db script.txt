create table carts(
	id int not null auto_increment,
    -- user_id int not null,
    user_id varchar(9) not null,
    creation_date date not null default now(),
    primary key(id),
    foreign key (user_id) references users(id)
);


create table products_in_cart(
	id int not null auto_increment,
    cart_id int not null,
    product_id int not null,
    amount decimal(7,3) not null default 1,-- int not null default 1, -- 
    cost decimal(7,2) not null,-- int not null, -- 
    primary key(id),
    foreign key (cart_id) references carts(id),
    foreign key (product_id) references products(id)
);


create table deliveries(
	id int not null auto_increment,
    cart_id int not null,
    total_cost decimal(7,2) not null, -- int not null, -- 
    delivery_city varchar(40) not null default '',
    delivery_address varchar(40) not null default '',
    delivery_date date not null default now(),
    credit varchar(4),
    creation_date datetime not null default now(),
    primary key(id),
    foreign key (cart_id) references carts(id)
);
